void makeNext(string p, int next[]) {
    int len = p.length();
    int j = 0, k = -1;
    next[0] = -1;
    while (j < len - 1) {
        if (k == -1 || p[j] == p[k]) {
            j++, k++;
            if (p[j] != p[k])
                next[j] = next[k];
            else
                k = next[j];
        } 
    }
}

int kmp(string s, string p) {
    int len1 = s.length();
    int len2 = p.length();
    int i = 0, j = 0;
    int *next = new int[len2];
    makeNext(p, next);
    while (i < len1 && j < len2) {
        if (j == -1 || s[i] == p[j])
            i++, j++;
        else
            j = next[j];
    }
    if (j == len2)
        return 1;
    else 
        return 0;
}
