void login()
{

	string account, password;
	int index = 0;
	char ch;
	while (1) {
		cout << "account: ";
		cin >> account;
		cout << "password: ";
		while ((ch = _getch()) != '\r') {
			if (ch == 8) {
				password[index] = '\0';
				break;
			}
			password += ch;
			index++;
			if (ch != '\b') {
				cout << "*";
			}
			else {
				if (index != 0) {
					cout << '\b' << " " << '\b';
					index--;
				}
			}
		}
		if ((account == "admin") && (password == "123456")) {
			break;
		}
		else {
			cout << "\nerror,check your account or password" << endl;
			system("pause");
			fflush(stdin);
			password.clear();
			system("cls");
		}
	}
}
